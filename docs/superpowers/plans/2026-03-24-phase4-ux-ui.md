  { name: '새 캠페인 회의 시작', action: 'new-meeting', keywords: ['meeting', '회의', '시작'] },
  { name: '일간 브리핑 실행', action: 'run-briefing', keywords: ['briefing', '브리핑'] },
  { name: 'GA4 분석 실행', action: 'run-ga4', keywords: ['ga4', '분석'] },
  { name: '마케팅 인텔 수집', action: 'run-collect', keywords: ['collect', '수집', '인텔'] },
];

export function CommandPalette() {
  const [open, setOpen] = useState(false);
  const router = useRouter();

  useEffect(() => {
    function onKeyDown(e: KeyboardEvent) {
      if (e.key === 'k' && (e.metaKey || e.ctrlKey)) {
        e.preventDefault();
        setOpen((o) => !o);
      }
    }
    document.addEventListener('keydown', onKeyDown);
    return () => document.removeEventListener('keydown', onKeyDown);
  }, []);

  const handleSelect = useCallback((value: string) => {
    setOpen(false);
    const page = PAGES.find((p) => p.path === value);
    if (page) {
      router.push(page.path);
      return;
    }
    // action handling
    if (value === 'new-meeting') router.push('/');
    if (value === 'run-briefing') {
      fetch('/api/jobs', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ jobId: 'daily-briefing' }) });
    }
    if (value === 'run-ga4') {
      fetch('/api/jobs', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ jobId: 'ga4-analysis' }) });
    }
    if (value === 'run-collect') {
      fetch('/api/jobs', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ jobId: 'collect-serper' }) });
    }
  }, [router]);

  if (!open) return null;

  return (
    <div className="command-palette-overlay" onClick={() => setOpen(false)}>
      <div className="command-palette-container" onClick={(e) => e.stopPropagation()}>
        <Command label="Garnet 커맨드 팔레트">
          <Command.Input placeholder="페이지 이동, 액션 실행..." autoFocus />
          <Command.List>
            <Command.Empty>결과 없음</Command.Empty>
            <Command.Group heading="페이지">
              {PAGES.map((page) => (
                <Command.Item key={page.path} value={page.path} onSelect={handleSelect} keywords={page.keywords}>
                  {page.name}
                </Command.Item>
              ))}
            </Command.Group>
            <Command.Group heading="액션">
              {ACTIONS.map((action) => (
                <Command.Item key={action.action} value={action.action} onSelect={handleSelect} keywords={action.keywords}>
                  {action.name}
                </Command.Item>
              ))}
            </Command.Group>
          </Command.List>
        </Command>
      </div>
    </div>
  );
}
```

// components/sparkline-card.tsx
'use client';

import { AreaChart, Area, ResponsiveContainer } from 'recharts';

interface SparklineCardProps {
  title: string;
  value: string | number;
  change?: number;      // % change vs previous period
  data?: { v: number }[];  // sparkline data points
  unit?: string;
}

export function SparklineCard({ title, value, change, data, unit }: SparklineCardProps) {
  const isPositive = (change ?? 0) >= 0;
  const changeColor = isPositive ? 'var(--status-success, #22c55e)' : 'var(--status-error, #ef4444)';

  return (
    <div className="metric-card" style={{ position: 'relative', overflow: 'hidden' }}>
      <div style={{ position: 'relative', zIndex: 1 }}>
        <p className="metric-label">{title}</p>
        <p className="metric-value">
          {value}{unit && <span style={{ fontSize: '0.6em', marginLeft: 4 }}>{unit}</span>}
        </p>
        {change !== undefined && (
          <p style={{ fontSize: 12, fontWeight: 600, color: changeColor, marginTop: 4 }}>
            {isPositive ? '+' : ''}{change.toFixed(1)}% vs 지난주
          </p>
        )}
      </div>
      {data && data.length > 0 && (
        <div style={{ position: 'absolute', bottom: 0, left: 0, right: 0, height: 48, opacity: 0.15 }}>
          <ResponsiveContainer width="100%" height="100%">
            <AreaChart data={data}>
              <Area
                type="monotone"
                dataKey="v"
                stroke="var(--accent, #3182F6)"
                fill="var(--accent, #3182F6)"
                strokeWidth={1.5}
              />
            </AreaChart>
          </ResponsiveContainer>
        </div>
      )}
    </div>
  );
}
```

```tsx
// components/morning-briefing-card.tsx
'use client';

import { useEffect, useState } from 'react';

interface DigestData {
  headline: string;
  summary: string;
  insights: { category: string; summary: string; source_count: number }[];
  actions: { priority: string; title: string }[];
  itemCount: number;
  createdAt: string;
}

export function MorningBriefingCard() {
  const [digest, setDigest] = useState<DigestData | null>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    fetch('/api/intel/digests?type=daily&limit=1')
      .then((r) => r.json())
      .then((data) => {
        const d = data.digests?.[0];
        if (d) {
          setDigest({
            headline: d.headline,
            summary: d.summary,
            insights: JSON.parse(d.insights || '[]'),
            actions: JSON.parse(d.actions || '[]'),
            itemCount: d.itemCount,
            createdAt: d.createdAt,
          });
        }
      })
      .catch(() => {})
      .finally(() => setLoading(false));
  }, []);

  if (loading) return null;
  if (!digest) {
    return (
      <div className="dashboard-hero" style={{ marginBottom: 24 }}>
        <p className="dashboard-eyebrow">오늘의 마케팅 브리핑</p>
        <p className="dashboard-copy">아직 수집된 인텔리전스가 없습니다. 워치리스트에 키워드를 추가하세요.</p>
      </div>
    );
  }

  const today = new Date(digest.createdAt).toLocaleDateString('ko-KR', { month: 'long', day: 'numeric', weekday: 'long' });

  return (
    <div className="dashboard-hero" style={{ marginBottom: 24 }}>
      <p className="dashboard-eyebrow">{today} 마케팅 브리핑</p>
      <h2 className="dashboard-title" style={{ fontSize: 20, marginBottom: 8 }}>{digest.headline}</h2>
      <p className="dashboard-copy" style={{ marginBottom: 16 }}>수집: {digest.itemCount}건</p>

      {digest.insights.length > 0 && (
        <div style={{ display: 'flex', gap: 12, flexWrap: 'wrap', marginBottom: 12 }}>
          {digest.insights.slice(0, 3).map((ins, i) => (
            <div key={i} className="soft-card" style={{ flex: '1 1 200px', padding: '10px 14px' }}>
              <span className="accent-pill" style={{ fontSize: 11, marginBottom: 4 }}>{ins.category}</span>
              <p style={{ fontSize: 13, marginTop: 4 }}>{ins.summary}</p>
            </div>
          ))}
        </div>
      )}

      {digest.actions.length > 0 && (
        <div style={{ display: 'flex', gap: 8, flexWrap: 'wrap' }}>
          {digest.actions.slice(0, 3).map((act, i) => (
            <span key={i} className={`status-badge ${act.priority === 'NOW' ? 'status-badge-warning' : 'status-badge-info'}`}>
              {act.priority}: {act.title}
            </span>
          ))}
        </div>
      )}
    </div>
  );
}
```

// app/intel/page.tsx
'use client';

import { useEffect, useState } from 'react';

interface IntelItem {
  id: string;
  platform: string;
  title: string;
  snippet: string;
  url: string;
  relevance: number;
  urgency: string;
  tags: string;
  createdAt: string;
}

const PLATFORM_LABELS: Record<string, string> = {
  SERPER: '웹/뉴스', NAVER: '네이버', YOUTUBE: 'YouTube',
  TWITTER: 'Twitter', REDDIT: 'Reddit',
};

const URGENCY_STYLES: Record<string, string> = {
  CRITICAL: 'status-badge-error',
  HIGH: 'status-badge-warning',
  NORMAL: 'status-badge-info',
  LOW: 'status-badge-neutral',
};

export default function IntelPage() {
  const [items, setItems] = useState<IntelItem[]>([]);
  const [platform, setPlatform] = useState('');
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    setLoading(true);
    const params = new URLSearchParams({ limit: '50' });
    if (platform) params.set('platform', platform);
    fetch(`/api/intel?${params}`)
      .then((r) => r.json())
      .then((data) => setItems(data.items || []))
      .catch(() => {})
      .finally(() => setLoading(false));
  }, [platform]);

  return (
    <div style={{ padding: '24px 32px', maxWidth: 900 }}>
      <h1 className="dashboard-title">마케팅 인텔리전스</h1>

      <div style={{ display: 'flex', gap: 8, marginBottom: 20, flexWrap: 'wrap' }}>
        <button className={`pill-option ${!platform ? 'pill-option-active' : ''}`} onClick={() => setPlatform('')}>전체</button>
        {Object.entries(PLATFORM_LABELS).map(([key, label]) => (
          <button key={key} className={`pill-option ${platform === key ? 'pill-option-active' : ''}`} onClick={() => setPlatform(key)}>
            {label}
          </button>
        ))}
      </div>

      {loading && <p className="dashboard-copy">로딩 중...</p>}

      <div style={{ display: 'flex', flexDirection: 'column', gap: 12 }}>
        {items.map((item) => {
          const tags: string[] = JSON.parse(item.tags || '[]');
          return (
            <a key={item.id} href={item.url} target="_blank" rel="noopener noreferrer" className="list-card" style={{ textDecoration: 'none' }}>
              <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'flex-start', gap: 12 }}>
                <div style={{ flex: 1 }}>
                  <div style={{ display: 'flex', gap: 6, marginBottom: 4 }}>
                    <span className="accent-pill" style={{ fontSize: 10 }}>{PLATFORM_LABELS[item.platform] || item.platform}</span>
                    <span className={`status-badge ${URGENCY_STYLES[item.urgency] || ''}`} style={{ fontSize: 10 }}>{item.urgency}</span>
                  </div>
                  <p style={{ fontWeight: 600, fontSize: 14, marginBottom: 4 }}>{item.title}</p>
                  <p style={{ fontSize: 13, color: 'var(--text-muted)', lineHeight: 1.5 }}>{item.snippet.slice(0, 200)}</p>
                  {tags.length > 0 && (
                    <div style={{ display: 'flex', gap: 4, marginTop: 6 }}>
                      {tags.slice(0, 4).map((tag, i) => (
                        <span key={i} style={{ fontSize: 11, padding: '2px 8px', borderRadius: 12, background: 'var(--surface-alt, #f5f5f5)' }}>{tag}</span>
                      ))}
                    </div>

                </div>
                <div style={{ textAlign: 'right', flexShrink: 0 }}>
                  <p style={{ fontSize: 11, color: 'var(--text-muted)' }}>{new Date(item.createdAt).toLocaleDateString('ko-KR')}</p>
                  <p style={{ fontSize: 11, color: 'var(--accent)', fontWeight: 600 }}>관련도 {(item.relevance * 100).toFixed(0)}%</p>
                </div>
              </div>
            </a>
          );
        })}
      </div>

      {!loading && items.length === 0 && (
        <div className="soft-card" style={{ textAlign: 'center', padding: 40 }}>
          <p className="dashboard-copy">수집된 인텔리전스가 없습니다.</p>
          <p style={{ fontSize: 13, color: 'var(--text-muted)' }}>워치리스트에 키워드를 등록하면 자동으로 수집이 시작됩니다.</p>
